const slides = document.querySelectorAll(".video-slide");
const titleLink = document.getElementById("video-title-link");
const descLink  = document.getElementById("video-desc-link");
const bar = document.getElementById("antik-bar");
const thumbPrev = document.getElementById("thumb-prev");
const thumbNext = document.getElementById("thumb-next");

let index = 0;

function showSlide(i) {
  slides.forEach((s, j) => {
    s.classList.toggle("active", j === i);
  });

  const current = slides[i];

  // Texto + href dinámicos para título y descripción
  titleLink.textContent = current.dataset.title || "";
  titleLink.href = current.dataset.titleUrl || "#";

  descLink.textContent = current.dataset.desc || "";
  descLink.href = current.dataset.descUrl || "#";

  // Miniaturas previa y siguiente
  const prev = slides[(i - 1 + slides.length) % slides.length];
  const next = slides[(i + 1) % slides.length];

  thumbPrev.style.backgroundImage = `url('${prev.dataset.img}')`;
  thumbPrev.querySelector("span").textContent = prev.dataset.title || "";

  thumbNext.style.backgroundImage = `url('${next.dataset.img}')`;
  thumbNext.querySelector("span").textContent = next.dataset.title || "";

  // Reinicio de barra
  bar.style.transition = "none";
  bar.style.width = "0%";
  setTimeout(() => {
    bar.style.transition = "width 10s ease-in-out";
    bar.style.width = "100%";
  }, 50);
}

function nextSlide() {
  index = (index + 1) % slides.length;
  showSlide(index);
}
function prevSlide() {
  index = (index - 1 + slides.length) % slides.length;
  showSlide(index);
}

function autoAdvance() {
  nextSlide();
  setTimeout(autoAdvance, 10000);
}

showSlide(0);
setTimeout(autoAdvance, 10000);
